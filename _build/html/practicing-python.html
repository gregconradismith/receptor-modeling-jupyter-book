# Format each element of the list with two decimal places and join them into a single string
def formatted_list(my_list,brackets=True):
    if brackets:
        f_list = '['
    else:
        f_list = ''
    for item in my_list:
        f_list = f_list+f' {item:.2f}'
    if brackets:
        f_list = f_list + ' ]'
    return f_list
